```kotlin
//--- counter fun create
private lateinit var counterTextView: TextView
private var counter = 1

fun incrementCounter(view: View) {  
    counter++  
    updateCounterText()  
}  
  
fun decrementCounter(view: View) {  
    if (counter > 0) {  
        counter--  
        updateCounterText()  
    }  
}  
private fun updateCounterText() {  
    counterTextView.text = "Counter: $counter"  
}

```
